# Quickshell Tileable Texture Subprocess Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a workflow-owned subprocess that generates, validates, and injects high-quality tileable/9-slice texture assets for ornate Quickshell `BorderImage` widget chrome.

**Architecture:** The craft pipeline will gain a focused texture asset module that runs before Quickshell QML generation when the design promises ornate/RPG/thorned chrome. The module creates a typed texture intent, writes deterministic local 9-slice PNG assets plus metadata, validates seam/tile quality, and injects asset paths/slice metadata into codegen so generated QML references real assets rather than placeholder rectangle borders.

**Tech Stack:** Python 3, Pillow, dataclasses, JSON metadata, QtQuick/QML `BorderImage`, existing linux-ricing craft pipeline, pytest.

```python
"""Tileable texture asset generation for ornate Quickshell BorderImage chrome."""
from __future__ import annotations

from dataclasses import dataclass, field
from pathlib import Path
import json
import re


_ORNATE_TERMS = (
    "ornate", "thorn", "thorns", "diablo", "rpg menu", "inventory frame",
    "carved", "carved border", "filigree", "relic", "blackiron", "bonfire",
    "9-slice", "tiled border", "borderimage",
)


@dataclass(frozen=True)
class TextureIntent:
    theme_slug: str
    palette: dict[str, str]
    material: str = "worn blackened iron, soot, parchment, ember-lit carved edges"
    variants: tuple[str, ...] = ("panel", "button", "slot", "plaque")
    forbidden_traits: tuple[str, ...] = (
        "plain rectangle",
        "flat orange outline",
        "copyrighted logo copy",
        "bulky terminal cage",
        "synthetic placeholder triangles",
    )
    slice_px: int = 28


@dataclass(frozen=True)
class TextureAsset:
    variant: str
    path: str
    slice_px: int
    width: int
    height: int
    seam_score: float


@dataclass(frozen=True)
class TextureBundle:
    root: Path
    assets: tuple[TextureAsset, ...]
    metadata_path: Path

    def as_prompt_context(self) -> str:
        data = {
            "root": str(self.root),
            "metadata": str(self.metadata_path),
            "assets": [asset.__dict__ for asset in self.assets],
        }
        return json.dumps(data, indent=2, sort_keys=True)


def _slug(value: str) -> str:
    value = value.strip().lower()
    value = re.sub(r"[^a-z0-9]+", "-", value)
    return value.strip("-") or "theme"


def _design_blob(design: dict) -> str:
    return json.dumps(design if isinstance(design, dict) else {}, sort_keys=True).lower()


def needs_texture_assets(element: str, design: dict) -> bool:
    if element != "widgets:quickshell":
        return False
    blob = _design_blob(design)
    if "quickshell" not in blob:
        return False
    return any(term in blob for term in _ORNATE_TERMS)


def extract_texture_intent(design: dict) -> TextureIntent:
    name = str(design.get("name") or design.get("theme_name") or "theme")
    palette = {
        str(k): str(v)
        for k, v in (design.get("palette") or {}).items()
        if isinstance(v, str) and v.startswith("#")
    }
    return TextureIntent(theme_slug=_slug(name), palette=palette)
```

```python
from PIL import Image, ImageDraw, ImageFilter


def _hex_to_rgba(value: str, alpha: int = 255) -> tuple[int, int, int, int]:
    value = value.lstrip("#")
    if len(value) != 6:
        return (214, 200, 170, alpha)
    return (int(value[0:2], 16), int(value[2:4], 16), int(value[4:6], 16), alpha)


def _palette_color(intent: TextureIntent, key: str, fallback: str, alpha: int = 255) -> tuple[int, int, int, int]:
    return _hex_to_rgba(intent.palette.get(key, fallback), alpha)


def _draw_wrapped_edge_teeth(draw: ImageDraw.ImageDraw, size: int, color: tuple[int, int, int, int], inset: int, step: int) -> None:
    # Use periodic placement so first/last segments align when repeated by BorderImage.
    for x in range(0, size + step, step):
        x0 = x % size
        draw.polygon([(x0, inset), ((x0 + step // 2) % size, inset + 9), ((x0 + step) % size, inset)], fill=color)
        y = size - inset
        draw.polygon([(x0, y), ((x0 + step // 2) % size, y - 9), ((x0 + step) % size, y)], fill=color)


def _draw_ornate_atlas(intent: TextureIntent, variant: str, size: int = 128) -> Image.Image:
    transparent = (0, 0, 0, 0)
    soot = _palette_color(intent, "bg", "#111513", 226)
    iron = _palette_color(intent, "iron", "#6b4b2f", 255)
    ember = _palette_color(intent, "ember", "#d06f22", 245)
    bone = _palette_color(intent, "bone", "#d6c8aa", 210)

    image = Image.new("RGBA", (size, size), transparent)
    draw = ImageDraw.Draw(image)
    b = intent.slice_px

    draw.rectangle([b, b, size - b, size - b], fill=soot)
    for offset, color, width in ((4, (0, 0, 0, 190), 6), (10, iron, 4), (17, bone, 1), (22, ember, 2)):
        draw.rectangle([offset, offset, size - offset - 1, size - offset - 1], outline=color, width=width)

    step = {"panel": 16, "button": 14, "slot": 12, "plaque": 18}.get(variant, 16)
    _draw_wrapped_edge_teeth(draw, size, ember, 8, step)

    # Distinct corners: hooks, pins, and bevel clusters. These must not just be stretched edges.
    for sx, sy, fx, fy in ((0, 0, 1, 1), (size, 0, -1, 1), (0, size, 1, -1), (size, size, -1, -1)):
        draw.polygon([(sx + fx * 10, sy + fy * 5), (sx + fx * 36, sy + fy * 11), (sx + fx * 18, sy + fy * 25)], fill=iron)
        draw.polygon([(sx + fx * 5, sy + fy * 10), (sx + fx * 11, sy + fy * 36), (sx + fx * 25, sy + fy * 18)], fill=iron)
        draw.ellipse([sx + fx * 22 - 5, sy + fy * 22 - 5, sx + fx * 22 + 5, sy + fy * 22 + 5], fill=ember)
        draw.ellipse([sx + fx * 22 - 2, sy + fy * 22 - 2, sx + fx * 22 + 2, sy + fy * 22 + 2], fill=bone)

    # Add subtle material grain, then sharpen; keep grain low to avoid noisy UI.
    grain = Image.effect_noise((size, size), 9).convert("L")
    alpha = grain.point(lambda p: 22 if p > 140 else 0)
    grain_rgba = Image.new("RGBA", (size, size), (214, 200, 170, 0))
    grain_rgba.putalpha(alpha)
    image = Image.alpha_composite(image, grain_rgba)
    return image.filter(ImageFilter.UnsharpMask(radius=1, percent=120, threshold=3))


def generate_texture_bundle(intent: TextureIntent, output_root: Path) -> TextureBundle:
    asset_root = output_root / "assets" / intent.theme_slug
    asset_root.mkdir(parents=True, exist_ok=True)
    assets: list[TextureAsset] = []
    for variant in intent.variants:
        image = _draw_ornate_atlas(intent, variant)
        rel = Path("assets") / intent.theme_slug / f"{variant}_9slice.png"
        path = output_root / rel
        image.save(path)
        seam_score = seam_distance(image)
        assets.append(TextureAsset(
            variant=variant,
            path=str(rel),
            slice_px=intent.slice_px,
            width=image.width,
            height=image.height,
            seam_score=seam_score,
        ))
    metadata_path = asset_root / "border_metadata.json"
    metadata_path.write_text(json.dumps({
        "theme_slug": intent.theme_slug,
        "slice_px": intent.slice_px,
        "assets": [asset.__dict__ for asset in assets],
    }, indent=2, sort_keys=True))
    return TextureBundle(root=asset_root, assets=tuple(assets), metadata_path=metadata_path)
```

```python
from PIL import ImageChops, ImageStat


def seam_distance(image: Image.Image, strip_px: int = 4) -> float:
    rgba = image.convert("RGBA")
    w, h = rgba.size
    left = rgba.crop((0, 0, strip_px, h))
    right = rgba.crop((w - strip_px, 0, w, h)).transpose(Image.Transpose.FLIP_LEFT_RIGHT)
    top = rgba.crop((0, 0, w, strip_px))
    bottom = rgba.crop((0, h - strip_px, w, h)).transpose(Image.Transpose.FLIP_TOP_BOTTOM)
    lr = ImageStat.Stat(ImageChops.difference(left, right)).mean
    tb = ImageStat.Stat(ImageChops.difference(top, bottom)).mean
    return float(sum(lr + tb) / (len(lr) + len(tb)))


def validate_texture_bundle(bundle: TextureBundle, max_seam_score: float = 55.0) -> list[str]:
    reasons: list[str] = []
    if not bundle.metadata_path.exists():
        reasons.append(f"missing texture metadata: {bundle.metadata_path}")
    for asset in bundle.assets:
        path = bundle.metadata_path.parent.parent.parent / asset.path
        if not path.exists():
            reasons.append(f"missing texture asset: {asset.path}")
            continue
        if asset.slice_px <= 0:
            reasons.append(f"invalid slice size for {asset.path}: {asset.slice_px}")
        if asset.seam_score > max_seam_score:
            reasons.append(f"texture asset {asset.path} seam score {asset.seam_score:.2f} exceeds {max_seam_score:.2f}")
    return reasons
```

```python
def _copy_texture_assets(research: dict, dest_dir: Path) -> list[str]:
    texture_assets = research.get("texture_assets") or {}
    root = Path(str(texture_assets.get("root", "")))
    copied: list[str] = []
    for asset in texture_assets.get("assets", []):
        rel = Path(str(asset.get("path", "")))
        if not rel or rel.is_absolute() or ".." in rel.parts:
            continue
        src = root.parent.parent / rel if root.name else Path()
        if not src.exists():
            # Fallback: if root is the asset directory, use basename.
            src = root / rel.name
        if not src.exists():
            continue
        dst = dest_dir / rel
        dst.parent.mkdir(parents=True, exist_ok=True)
        dst.write_bytes(src.read_bytes())
        copied.append(str(dst))
    return copied
```
